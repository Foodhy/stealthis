Here's how you can implement a debounce function in JavaScript:

function debounce(fn, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

// Usage
const onSearch = debounce((query) => {
  console.log('Searching:', query);
}, 300);

The function delays execution until delay milliseconds have passed since the last call. It's ideal for search inputs, resize handlers, and scroll events.
