Key File — src/index.ts
import { Hono } from "hono";
import { logger } from "hono/logger";
import { cors } from "hono/cors";

const app = new Hono();

app.use("*", logger());
app.use("/api/*", cors());

app.get("/", (c) => c.json({ message: "Hello Hono!" }));
app.get("/api/health", (c) => c.json({ status: "ok" }));

export default app;
